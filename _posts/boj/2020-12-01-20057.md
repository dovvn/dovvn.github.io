# 문제

- [BOJ 20057](https://www.acmicpc.net/problem/20057)
- 문제의 저작권은 Baekjoon Online Judge에 있습니다.

# 접근방식

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

//마법사 상어와 토네이도
public class Main {

	static int N; //격자 크기
	static int[][] map;//전체 맵
	static int nowX, nowY; //현재 좌표
	static int nextX, nextY; //다음 좌표
	static int go=1; //이동 횟수
	static int[] di = {0, 1, 0, -1};
	static int[] dj = {-1, 0, 1, 0};
	static int dir; //현재 방향
	static int count;
	static int Ans; //밖으로 나간 모래 합

	public static void main(String[] args) throws NumberFormatException, IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		N = Integer.parseInt(br.readLine());
		map = new int[N+1][N+1]; //(1,1)부터 시작

		for(int i=1; i<=N; i++) {
			StringTokenizer st = new StringTokenizer(br.readLine());
			for(int j=1; j<=N; j++) {
				map[i][j] = Integer.parseInt(st.nextToken());
			}
		}//input 모래양

		//시작 좌표
		nowX = (N+1)/2;
		nowY = (N+1)/2;
		map[nowX][nowY] = dir;

		boolean isGo = true;

		//달팽이숫자
		while(isGo) {
			for(int i=0; i<2; i++) { //2번 반복
				for(int j=1; j<=go; j++) { //go번 앞으로 감
					nextX = nowX + di[dir];
					nextY = nowY + dj[dir];
					tornado(dir);
					nowX = nextX;
					nowY = nextY;
				}
				dir = (dir+1)%4; //다음 방향으로
				if(nowX == 1 && nowY == N) {
					for(int z=1; z<=go; z++) {
						nextX = nowX + di[dir];
						nextY = nowY + dj[dir];
						tornado(dir);
						nowX = nextX;
						nowY = nextY;
					}
					isGo = false; //종료
					break;
				}
			}
			go++;
		}
		System.out.println(Ans);
	}

	//모래 뿌리기
	private static void tornado(int dir) {
		int now = map[nextX][nextY];
		//1. 각 비율
		int s1 = (int)(now * 0.01);
		int s2 = (int)(now * 0.02);
		int s5 = (int)(now * 0.05);
		int s7 = (int)(now * 0.07);
		int s10 = (int)(now * 0.1);

		//2. 남은 모래
		int a = now - 2*(s1+s2+s7+s10)-s5;

		map[nextX][nextY] = 0;

		//3. 각 비율로 모래가 갈 수 있는지 판단
		if(dir == 0 || dir == 2) {
			//1%
			for(int i = 0; i < 2; i++) {
				int sx = nowX + di[1+2*i];
	            int sy = nowY + dj[1+2*i];
	            check(sx, sy, s1);
	        }

			//2%
			for(int i = 0; i < 2; i++) {
				int sx = nextX + di[1+2*i]*2;
	            int sy = nextY + dj[1+2*i]*2;
	            check(sx, sy, s2);
	        }

			//7%
			for(int i = 0; i < 2; i++) {
				int sx = nextX + di[1+2*i];
	            int sy = nextY + dj[1+2*i];
	            check(sx, sy, s7);
	        }

			//10%
			for(int i = 0; i < 2; i++) {
				int sx = nextX + di[dir] + di[1+2*i];
	            int sy = nextY + dj[dir] + dj[1+2*i];
	            check(sx, sy, s10);
	        }

			//5%
			int sx = nextX + di[dir]*2;
	        int sy = nextY + dj[dir]*2;
	        check(sx, sy, s5);

	        //a(알파)
	        sx = nextX + di[dir];
	        sy = nextY + dj[dir];
	        check(sx, sy, a);
		}else {
			//1%
			for(int i = 0; i < 2; i++) {
				int sx = nowX + di[2*i];
	            int sy = nowY + dj[2*i];
	            check(sx, sy, s1);
	        }

			//2%
			for(int i = 0; i < 2; i++) {
				int sx = nextX + di[2*i]*2;
	            int sy = nextY + dj[2*i]*2;
	            check(sx, sy, s2);
	        }

			//7%
			for(int i = 0; i < 2; i++) {
				int sx = nextX + di[2*i];
	            int sy = nextY + dj[2*i];
	            check(sx, sy, s7);
	        }

			//10%
			for(int i = 0; i < 2; i++) {
				int sx = nextX + di[dir] + di[2*i];
	            int sy = nextY + dj[dir] + dj[2*i];
	            check(sx, sy, s10);
	        }

			//5%
			int sx = nextX + di[dir]*2;
	        int sy = nextY + dj[dir]*2;
	        check(sx, sy, s5);

	        //a(알파)
	        sx = nextX + di[dir];
	        sy = nextY + dj[dir];
	        check(sx, sy, a);
		}
	}

	private static void check(int sx, int sy, int s) {
		if(sx>=1 && sy>=1 && sx<=N && sy<=N) {
			map[sx][sy] += s;
		}else {
			Ans += s;
		}
	}
}
```
